import { WalletPay } from '@tonconnect/sdk';

const createOrder = async (items: CartItem[], shippingAddress: Address) => {
  const totalTON = calculateTotal(items);

  const order = await walletPay.createOrder({
    amount: totalTON,
    currency: 'TON',
    description: `MemeMart Order #${orderId}`,

    returnUrl: `https://mememart.app/order/${orderId}`,

    }
  });

## MemeSeal Integration

### Every Product = Sealed

When a product ships:
1. Generate unique product ID
2. Create MemeSeal timestamp with:
   - Product SKU
   - Serial number (if applicable)
   - Buyer (optional, for warranty)
   - Purchase date
   - Transaction hash
3. Send seal to buyer
4. Buyer can verify authenticity forever

### Use Cases

**Resale verification**: "This Flipper Zero is authentic, here's the seal"
**Warranty claims**: Prove purchase date without receipt
**Collectible value**: Limited edition items with provable scarcity

## Legal Considerations

### Product Disclaimers

Every product page must include:
- "For authorized security testing and educational purposes only"
- "Check local laws before purchasing and using"
- "Buyer assumes all responsibility for legal compliance"

### Restricted Countries

Some products may not ship to:
- Countries with strict import controls on electronics
- Regions with RF transmission restrictions

### Compliance

- All products sourced from authorized channels (no grey market)
- No "hacking tools" that are illegal to possess (vary by jurisdiction)
- Clear terms of service
- Age verification (18+)

## Go-to-Market

### Phase 1: MVP (Week 1-4)

**Products**:
- 3 DIY kits (StingRay, P4wnP1, WiFi Deauther)
- Merch (print-on-demand)

**Why start here**:
- Highest margin
- No supplier relationships needed initially
- Test demand before bigger investment

### Phase 2: Authorized Products (Month 2-3)

**Add**:
- Flipper Zero (Lab401 partnership)
- Hak5 essentials (Rubber Ducky, Bash Bunny)

**Actions**:
- Apply to reseller programs
- Negotiate drop-ship arrangements
- Build inventory for popular items

### Phase 3: Full Catalog (Month 4+)

**Add**:
- Full Hak5 line
- Hardware wallets
- Advanced kits
- Limited edition drops
